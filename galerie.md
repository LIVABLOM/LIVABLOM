<script>
  function filterGallery(category) {
    const items = document.querySelectorAll('.gallery-item');
    items.forEach(item => {
      if (category === 'all') {
        item.style.display = 'block';
      } else {
        item.style.display = item.classList.contains(category) ? 'block' : 'none';
      }
    });
  }
</script>
